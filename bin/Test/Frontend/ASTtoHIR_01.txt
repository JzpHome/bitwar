{ Sample program
  in TINY language -
  computes factorial
}
{ read x; { input an integer }
Stategy T1:
	A[CUR+1] := A[CUR];
	if 0 < x then { don't compute if x <= 0 }
		fact := 1;
		while x > 0 do 
			fact := fact * x;
			x := x - 1
		end
		   { write fact  { output factorial of x }
	end
end
